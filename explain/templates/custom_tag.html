--Custom Tag-- Create templatetags folder in your app. Create file there and name it whatever you want. for example: blog_tags.py insert following lines in the top. from django import template register = template.Library() add your first tag. 2 steps: register your tag name add your tag function Examples 1. calculates the sum of a list of numbers. @register.simple_tag(name="sum_list") def sum_list(my_list): return sum(my_list) 2. multiplies two numbers. @register.simple_tag(name="multiply") def multiply(a, b): return a * b 3. capitalizes the first letter of each word in a string. @register.simple_tag(name="capitalize_words") def capitalize_words(value): return ' '.join([word.capitalize() for word in value.split()]) 4. generates a random number within a specified range. @register.simple_tag(name="simple_tag") def random_number(min_value, max_value): return random.randint(min_value, max_value) 5. gets the full name of a user from their first and last names. @register.simple_tag(name="user_full_name") def user_full_name(user): return f"{user.first_name} {user.last_name}" use those filters in your templates. {% load blog_tags %} 1. {% sum_list my_numbers_list %} input: [1, 3, 5, 7, 9] # 1+3+5+7+9 output: 25 2. {% multiply 5 3 %} # 5*3 output: 15 3. {% capitalize_words "hello world" %} output: Hello World 4. {% random_number 1 4 %} output: could be 1, 2, 3, 4 5. {% user_full_name user_object %} # user.first_name = khaled, user.last_name = alshami output: khaled alshami
